#include #include #include // Настройка SoftwareSerial #define RX_PIN 2 // Подключите к TX датчика #define TX_PIN 3 // Подключите к RX датчика SoftwareSerial mySerial(RX_PIN, TX_PIN); MHZ19 myMHZ19; void setup() { Serial.begin(9600); mySerial.begin(9600); myMHZ19.begin(mySerial); myMHZ19.autoCalibration(false); // Отключаем авто-калибровку Serial.println("MH-Z19B Initializing..."); // Ждем стабилизации датчика delay(2000); // Проверяем соединение if (myMHZ19.getCO2() == 0) { Serial.println("Warning: Check sensor connection!"); } else { Serial.println("Sensor OK!"); } } void loop() { // Получаем значение CO2 int co2 = myMHZ19.getCO2(); // Получаем температуру int temp = myMHZ19.getTemperature(); // Получаем статус калибровки int calStatus = myMHZ19.getStatus(); Serial.println("========== MH-Z19B =========="); Serial.print("CO2: "); Serial.print(co2); Serial.println(" ppm"); Serial.print("Temperature: "); Serial.print(temp); Serial.println(" °C"); // Расшифровка статуса Serial.print("Calibration Status: "); switch(calStatus) { case 0: Serial.println("OK"); break; case 1: Serial.println("Warming up"); break; case 2: Serial.println("Calibration needed"); break; default: Serial.println("Unknown"); } // Анализ качества воздуха Serial.print("Air Quality: "); if (co2 < 600) { Serial.println("Excellent"); } else if (co2 < 1000) { Serial.println("Good"); } else if (co2 < 1500) { Serial.println("Moderate"); } else if (co2 < 2000) { Serial.println("Poor"); } else { Serial.println("Bad - Ventilate immediately!"); } Serial.println("============================="); // Задержка между измерениями delay(5000); }
